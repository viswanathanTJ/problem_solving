            System.out.println(s);

            mi[i] = new Interval(start, end);
        }
        Arrays.sort(ni, Comparator.comparingLong(b -> b.start));
        Arrays.sort(mi, Comparator.comparingLong(b -> b.start));
        long ans = 0;
        int i = 0, j = 0;
        while (i < n && j < m) {
            long l = max(ni[i].start, mi[j].start);
            long r = min(ni[i].end, mi[j].end);

            if (l <= r)
                ans += r - l;

            if (ni[i].end < mi[j].end)
                i++;
            else
                j++;
        }
        System.out.println(ans);
    }

    static long max(long a, long b) {
        return a > b ? a : b;
    }

    static long min(long a, long b) {
        return a < b ? a : b;
    }

    static class Interval {
        long start;
        long end;
        Interval(long s, long e) {
            start = s;
            end = e;
        }
    }

{

import java.util.*;
import java.lang.*;
import java.io.*;

/* Name of the class has to be "Main" only if the class is public. */
class Codechef
{
    static boolean checkSorted(int n, String arr) {
        char[] b = new char[n];

        for(int i=0;i<n;i++)
            b[i] = arr.charAt(i);
        Arrays.sort(b, 0, n);
        int ct = 0;
        for (int i = 0; i < n; i++)
            if (arr.charAt(i) != b[i])
                ct++;
        if (ct == 0 || ct == 2)
            return true;
        else
            return false;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int t,n;
        String s;
        t = sc.nextInt();
        while(t-->0) {
            n = sc.nextInt();
            s = sc.next();
            if (checkSorted(n, s))
                System.out.println("YES");
            else
               System.out.println("NO");
        }
    }
}
